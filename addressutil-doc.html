addressutil

Contains various utilities for handling addresses in the United States. Requires Python 3.8 or newer.

Quick start

To quickly get the cleaned-up version of an address, you can do

>>> import addressutil
>>> str(addressutil.Address.parse("123 MAIN ST.", "SPRINGFIELD IL 12345-6789"))
'123 MAIN ST
SPRINGFIELD IL 12345-6789'

You can also work with Address objects directly, which are comparable. This is the preferred way to represent addresses.

>>> import addressutil
>>> addressutil.Address.parse("123 N.W. MAIN ST.", "SPRINGFIELD IL 12345-6789") == addressutil.Address.parse("123 NW MAIN STREET", "SPRINGFIELD IL 12345-6789")
True

The function addressutil.Address.parse() takes two arguments: the delivery address itself and the last line of the address (where the city, state, and ZIP code are usually written). The outputted string separates the delivery address and the last line with a newline, just like how it would be written on an envelope.

 49class AddressParseError(RuntimeError):
 50    """
 51    Raised when the parsing of an address string failed.
 52    """
 53    def __init__(self, description: str, address_string: str, index: int):
 54        """
 55        :param description: A brief description of the issue.
 56        :param address_string: Original address string, to provide a reference as to the location of the error.
 57        :param index: The index in the address string at which the parse error occured.
 58        """
 59        output_string = "Error parsing address at index " + str(index) + " (" + description + "), near:\n"
 60        output_string += address_string + '\n'
 61        output_string += (' ' * index) + "^\n"
 62        super().__init__(output_string)
 63        self.index = index
 64    
 65    def __reduce__(self):
 66        # Make the exception pickleable, just in case,
 67        return (AddressParseError, (self.message, self.index))

 70class _AddressToken:
 71    """
 72    Represents a token used during the parsing of an address. Should not be accessed outside
 73    of the addressutil module, unless you really know what you are doing.
 74    """
 75    TYPE_DECIMAL = "decimal"
 76    """Represents a decimal number"""
 77    TYPE_FRACTION = "fraction"
 78    """Represents a fractional number"""
 79    TYPE_NUMBER = "number"
 80    """Represents a nonnegative integer"""
 81    TYPE_NUMBERSUFFIXED = "numbersuffixed"
 82    """Represents a number suffixed with a non-numeric character"""
 83    TYPE_POUND = "pound"
 84    """Represents the pound sign (#)"""
 85    TYPE_SPECIAL = "special"
 86    """Represents a token that begins with a nonalphanumeric character"""
 87    TYPE_WORD = "word"
 88    """Represents a alphanumeric word beginning with a letter"""

 91        """
 92        :param literal: The literal value of the token.
 93        :param type_: The type of the token.

 95        """
 96        self.literal: str = literal
 97        self.type_: str = type_
 98        self.index: int = index

171    @staticmethod
172    def _lex_delivery_address(delivery_address: str) -> list[_AddressToken]:
173        """
174        Internal function to process a string into a series of tokens.
175        
176        :return: List of Address tokens
177        """
178        output_tokens: list[_AddressToken] = []
179        
180        current_position: int = 0  # The character we are currently on within the delivery address
181        current_token: Optional[_AddressToken] = None
182        while current_position < len(delivery_address):
183            if delivery_address[current_position] == "\n":
184                raise AddressParseError("unexpected newline", delivery_address, current_position)
185            if current_token is None:
186                # No current token.
187                if re.match(r"[^A-Za-z0-9#\-\.\/]", delivery_address[current_position]):
188                    # We encountered another non-parsed character. Do nothing.
189                    pass
190                elif re.match(r"\d", delivery_address[current_position]):
191                    # Begins with a number. Assume it is a number for now, although it could change.
192                    current_token = _AddressToken(delivery_address[current_position], _AddressToken.TYPE_NUMBER, current_position)
193                elif re.match(r"[A-Za-z]", delivery_address[current_position]):
194                    # Begins with alphabetic character
195                    current_token = _AddressToken(delivery_address[current_position], _AddressToken.TYPE_WORD, current_position)
196                elif re.match(r"#", delivery_address[current_position]):
197                    # The pound sign is a special token that will only ever consist of a single pound sign.
198                    output_tokens.append(_AddressToken(delivery_address[current_position], _AddressToken.TYPE_POUND, current_position))
199                else:
200                    # Begins with hyphen, period, or forward slash,
201                    current_token = _AddressToken(delivery_address[current_position], _AddressToken.TYPE_SPECIAL, current_position)
202            else:
203                if re.match(r"[^A-Za-z0-9\-\.\/]", delivery_address[current_position]):
204                    # We encountered a non-parsed character. Terminate the current token.
205                    output_tokens.append(current_token)
206                    current_token = None
207                    
208                    # Also terminate the token if we encounter a pound (#) sign.
209                    # However, we also want to add the pound sign in as a separate token.
210                    if re.match(r"#", delivery_address[current_position]):
211                        # Special pound sign token.
212                        output_tokens.append(_AddressToken(delivery_address[current_position], _AddressToken.TYPE_POUND, current_position))
213                else:
214                    # Append whatever character we have on hand to the previous token.
215                    if current_token.type_ == _AddressToken.TYPE_NUMBER:
216                        if re.match(r"\.", delivery_address[current_position]):
217                            # A dot indicates a decimal
218                            current_token.type_ = _AddressToken.TYPE_DECIMAL
219                        elif re.match(r"\/", delivery_address[current_position]):
220                            # A slash indicates a fraction
221                            current_token.type_ = _AddressToken.TYPE_FRACTION
222                        elif re.match(r"[^0-9]", delivery_address[current_position]):
223                            # This is no longer a number, but a suffixed number
224                            current_token.type_ = _AddressToken.TYPE_NUMBERSUFFIXED
225                    elif current_token.type_ == _AddressToken.TYPE_DECIMAL or current_token.type_ == _AddressToken.TYPE_FRACTION:
226                        if re.match(r"[^0-9]", delivery_address[current_position]):
227                            # This is not a decimal or a fraction, it is now a generic suffixed number.
228                            current_token.type_ = _AddressToken.TYPE_NUMBERSUFFIXED
229                    current_token.literal += delivery_address[current_position]
230            current_position += 1
231        # Append the last token in the string, if any.
232        if current_token is not None:
233            output_tokens.append(current_token)
234        return output_tokens

236    @staticmethod
237    def parse(delivery_address: str, last_line: str) -> Address:
238        """
239        Parses an address from a string. Requires delivery_address and last_line to be separated. Usually, these
240        are written as separate lines on an envelope. For example, consider the following written on an envelope:
241        ```
242        JOHN SMITH
243        1003 S MAIN ST UNIT 603
244        SPRINGFIELD IL 84785-9634
245        ```
246        The first line is discarded, as it is not part of the of the address but rather the recipient name. The
247        second line, the delivery address, is 1003 S MAIN ST UNIT 603. Pass this in for the delivery_address
248        parameter. The last line (SPRINGFIELD IL 84785) should be passed in for the last_time parameter. Note
249        that this function only parses the address, it does **not** check if the address is valid or deliverable.
250        
251        With the exception of newlines, unknown characters are treated as spaces during parsing. There should
252        not be any newlines in either parameter.
253        
254        :param delivery_address: The delivery address (line immediately under the recipient name).
255        :param last_line: The last line of the address, usually containing city, state, and ZIP.
256        :return: A subclass of Address.
257        :raises AddressParseError: If there was an issue with parsing the delivery address.
258        :raises ValueError: If there was an issue with parsing the address last line.
259        """
260        # Parse city, state, and zip first. Then, parse the delivery address itself.
261        # All non-alphanumeric characters will be silently ignored.
262        last_line = last_line.upper().strip()
263        last_line = re.sub(r"\s", " ", last_line)
264        last_line = re.sub(r"[^A-Za-z0-9\s\-]", "", last_line)
265        
266        last_line_tokens: list[str] = last_line.split(' ')
267        city: str = ""
268        state: str = ""
269        zipcode: str = ""
270        zipcode_ext: Optional[str] = ""
271        if len(last_line_tokens) <= 1:
272            raise ValueError("Missing 2-character state code in address last line \"" + last_line + "\"")
273        if len(last_line_tokens) <= 2:
274            raise ValueError("Missing city in address last line \"" + last_line + "\"")
275        
276        # Now parse the tokens
277        if '-' in last_line_tokens[-1]:
278            # ZIP+4
279            zipcode = last_line_tokens[-1].split('-')[0]
280            zipcode_ext = last_line_tokens[-1].split('-')[-1]
281            if re.search(r"[^0-9]", zipcode) or re.search(r"[^0-9]", zipcode_ext):
282                raise ValueError("Invalid characters in ZIP+4 code \"" + last_line_tokens[-1] + "\"")
283            if len(zipcode) != 5 or len(zipcode_ext) != 4:
284                raise ValueError("ZIP+4 code \"" + last_line_tokens[-1] + "\" has the wrong length. All ZIP codes must be zero-padded.")
285        else:
286            # ZIP
287            zipcode = last_line_tokens[-1]
288            zipcode_ext = None
289            if re.search(r"[^0-9]", zipcode):
290                raise ValueError("Invalid characters in ZIP code \"" + last_line_tokens[-1] + "\"")
291            if len(zipcode) != 5:
292                raise ValueError("ZIP code \"" + last_line_tokens[-1] + "\" has the wrong length. All ZIP codes must be zero-padded.")
293        if len(last_line_tokens[-2]) != 2:
294            raise ValueError("Invalid 2-character state code \"" + last_line_tokens[-2] + "\". The 2-character abbreviation should " + 
295                    "be used instead of the full state name.")
296        state = last_line_tokens[-2]
297        city = " ".join(last_line_tokens[:-2])
298        
299        # At this point, the variables city, state, zipcode, and zipcode_ext should be populated. We now turn to parsing the delivery address.
300        
301        unparsed_tokens: list[_AddressToken] = Address._lex_delivery_address(delivery_address.upper().strip())
302        """List of tokens not yet parsed."""
303        
304        
305        # Make sure we have at least 1 token to work with.
306        if len(unparsed_tokens) == 0:
307            raise AddressParseError("no valid tokens found", delivery_address, 0)
308        
309        if len(unparsed_tokens) == 2 and unparsed_tokens[0].literal == "GENERAL" and unparsed_tokens[1].literal == "DELIVERY":
310            # General delivery
311            return GeneralDeliveryAddress(city, state, zipcode, zipcode_ext)
312        elif re.sub(r"[^A-Za-z0-9]", "", unparsed_tokens[0].literal) in {"HC", "RR", "CPR", "OPC", "PSC", "UPR", "UNIT"} and (len(unparsed_tokens) >= 3 and unparsed_tokens[2].literal == "BOX"):
313            # TODO Make sure "RR03 BOX 98D" is correctly recognized as a Rural Route address.
314            route_address_type = "Highway Contract Route" if unparsed_tokens[0].literal == "HC" else "Rural Route"
315            if unparsed_tokens[0].literal not in {"HC", "RR"}:
316                route_address_type = "Overseas Military"
317            if len(unparsed_tokens) < 4:
318                raise AddressParseError("missing box number in " + route_address_type.lower() + " address", delivery_address, unparsed_tokens[-1].index + len(unparsed_tokens[-1].literal))
319            if unparsed_tokens[1].type_ not in {_AddressToken.TYPE_NUMBER, _AddressToken.TYPE_NUMBERSUFFIXED}:
320                raise AddressParseError("invalid " + route_address_type.lower() + " number, cannot begin with a letter or symbol", delivery_address, unparsed_tokens[1].index)
321            route_number: str = unparsed_tokens[1].literal
322            box_number: str = unparsed_tokens[3].literal
323            if unparsed_tokens[0].literal == "HC":
324                return HighwayContractRouteAddress(route_number, box_number, city, state, zipcode, zipcode_ext)
325            elif unparsed_tokens[0].literal == "RR":
326                return RuralRouteAddress(route_number, box_number, city, state, zipcode, zipcode_ext)
327            else:
328                # In the case of overseas military, the "route_number" is the address number.
329                return OverseasMilitaryAddress(unparsed_tokens[0].literal, route_number, box_number, city, state, zipcode, zipcode_ext)
330        elif len(unparsed_tokens) >= 2 and re.sub(r"[^A-Za-z0-9]", "", unparsed_tokens[0].literal) == "PO" and unparsed_tokens[1].literal == "BOX":
331            if len(unparsed_tokens) < 3:
332                raise AddressParseError("missing box number in post office box address", delivery_address, unparsed_tokens[-1].index + len(unparsed_tokens[-1].literal))
333            po_box_number: str = unparsed_tokens[2].literal
334            if po_box_number[0] == '-':
335                po_box_number = "0" + po_box_number[1:]
336            return PostOfficeBoxAddress(po_box_number, city, state, zipcode, zipcode_ext)
337        
338        # Assuming standard address
339        
340        address_number: Optional[str] = None
341        predirectional: Optional[str] = None
342        street_name: Optional[str] = None
343        street_suffix: Optional[str] = None
344        postdirectional: Optional[str] = None
345        address2_type: Optional[str] = None
346        address2: Optional[str] = None
347        
348        # First, get the address number, if it exists.
349        if len(unparsed_tokens) >= 2 and \
350                unparsed_tokens[0].type_ in {_AddressToken.TYPE_NUMBER, _AddressToken.TYPE_FRACTION, _AddressToken.TYPE_DECIMAL} or \
351                (_AddressToken.TYPE_NUMBERSUFFIXED and "-" in unparsed_tokens[0].literal):
352            address_number = unparsed_tokens[0].literal
353            if unparsed_tokens[0].type_ == _AddressToken.TYPE_NUMBER and unparsed_tokens[1].type_ == _AddressToken.TYPE_FRACTION:
354                address_number += " " + unparsed_tokens[1].literal
355                unparsed_tokens.pop(0)
356            unparsed_tokens.pop(0)
357        
358        # At this point, the address number has been popped. No need to worry about it again.
359        
360        # Find where the suffix and secondary address identifier are.
361        suffix_index: Optional[int] = None
362        address2_index: Optional[int] = None
363        already_seen_hashtag: bool = False  # Whether a hashtag has already been found.
364        for i in range(len(unparsed_tokens)):
365            current_token = unparsed_tokens[i]
366            if current_token.literal.replace('.', "") in SECONDARY_UNIT_INDICATORS:
367                address2_index = i
368            if current_token.literal.replace('.', "") in STREET_SUFFIXES:
369                suffix_index = i
370            if current_token.type_ == _AddressToken.TYPE_POUND:
371                if already_seen_hashtag:
372                    raise AddressParseError("only at most one pound sign (#) is permitted in an address", delivery_address, current_token.index)
373                if address2_index is not None and i == address2_index + 1:
374                    raise AddressParseError("cannot have pound sign (#) if unit specifier is already present", delivery_address, current_token.index)
375                address2_index = i
376                already_seen_hashtag = True
377            if current_token.type_ == _AddressToken.TYPE_SPECIAL:
378                raise AddressParseError("unexpected token", delivery_address, current_token.index)
379        
380        # The secondary address must be after the street suffix.
381        if suffix_index is not None and address2_index is not None and address2_index <= suffix_index:
382            address2_index = None
383        
384        # Parse the secondary address and pop it from the list, if it exists
385        if address2_index is not None:
386            address2_type = unparsed_tokens[address2_index].literal
387            if address2_index == len(unparsed_tokens) - 1:
388                raise AddressParseError("missing number for secondary address", delivery_address, unparsed_tokens[-1].index + len(unparsed_tokens[-1].literal))
389            address2 = " ".join((x.literal for x in unparsed_tokens[address2_index+1:]))
390            
391            # Keep removing unparsed tokens until we have reached the index of address2.
392            while len(unparsed_tokens) > address2_index:
393                unparsed_tokens.pop()
394        
395        # Now, we have the predirectional, street name, and street suffix, and postdirectional.
396        
397        # Check if we have a postdirectional.
398        if len(unparsed_tokens) > 0 and unparsed_tokens[-1].literal.replace('.', "") in DIRECTIONS_LIST:
399            # Yes, there is a postdirectional! Indicate its presence and pop it from the list.
400            postdirectional = unparsed_tokens[-1].literal
401            unparsed_tokens.pop()
402        
403        # At this point, only the predirectional, street name, and street suffix should be in the list, if they exist at all.
404        
405        # Now check the location of the suffix. If it's not at the end, the "suffix" we identified was actually not the true suffix.
406        if suffix_index is not None and len(unparsed_tokens) - suffix_index > 1:
407            # Okay, the "suffix" that we expected turned out not the be the suffix. That's okay, let's mark it as not a suffix.
408            suffix_index = None
409        
410        # TODO Standardize highway names (like COUNTY ROAD 238A, INTERSTATE 75, etc.)
411        
412        if suffix_index is None:
413            # The starting and ending index of the street name. If there is a postdirectional, don't count that here.
414            if len(unparsed_tokens) > 1 and unparsed_tokens[0].literal.replace('.', "") in DIRECTIONS_LIST:
415                predirectional = unparsed_tokens[0].literal
416                unparsed_tokens.pop(0)
417            street_name = " ".join((x.literal for x in unparsed_tokens))
418        else:
419            street_start_index = 0
420            if suffix_index > 1:
421                if unparsed_tokens[0].literal.replace('.', "") in DIRECTIONS_LIST:
422                    predirectional = unparsed_tokens[0].literal
423                    street_start_index = 1
424            street_suffix = unparsed_tokens[suffix_index].literal
425            street_name = " ".join((x.literal for x in unparsed_tokens[street_start_index:suffix_index]))
426        
427        # More standardization of the address.
428        if predirectional is not None:
429            predirectional = re.sub(r"[^A-Za-z]", "", predirectional)
430        if postdirectional is not None:
431            postdirectional = re.sub(r"[^A-Za-z]", "", postdirectional)
432        if street_suffix is not None:
433            street_suffix = re.sub(r"[^A-Za-z]", "", street_suffix)
434        predirectional = DIRECTIONS_MAPPING[predirectional] if predirectional in DIRECTIONS_MAPPING else predirectional
435        postdirectional = DIRECTIONS_MAPPING[postdirectional] if postdirectional in DIRECTIONS_MAPPING else postdirectional
436        street_suffix = STREET_SUFFIX_MAPPING[street_suffix] if street_suffix in STREET_SUFFIX_MAPPING else street_suffix
437        
438        # Hyphens will be replaced with spaces by USPS as well.
439        street_name = street_name.replace('-', ' ')
440        
441        return StandardAddress(address_number, predirectional, street_name, street_suffix, postdirectional, address2_type, address2,
442                city, state, zipcode, zipcode_ext)

173    def _lex_delivery_address(delivery_address: str) -> list[_AddressToken]:
174        """
175        Internal function to process a string into a series of tokens.
176        
177        :return: List of Address tokens
178        """
179        output_tokens: list[_AddressToken] = []
180        
181        current_position: int = 0  # The character we are currently on within the delivery address
182        current_token: Optional[_AddressToken] = None
183        while current_position < len(delivery_address):
184            if delivery_address[current_position] == "\n":
185                raise AddressParseError("unexpected newline", delivery_address, current_position)
186            if current_token is None:
187                # No current token.
188                if re.match(r"[^A-Za-z0-9#\-\.\/]", delivery_address[current_position]):
189                    # We encountered another non-parsed character. Do nothing.
190                    pass
191                elif re.match(r"\d", delivery_address[current_position]):
192                    # Begins with a number. Assume it is a number for now, although it could change.
193                    current_token = _AddressToken(delivery_address[current_position], _AddressToken.TYPE_NUMBER, current_position)
194                elif re.match(r"[A-Za-z]", delivery_address[current_position]):
195                    # Begins with alphabetic character
196                    current_token = _AddressToken(delivery_address[current_position], _AddressToken.TYPE_WORD, current_position)
197                elif re.match(r"#", delivery_address[current_position]):
198                    # The pound sign is a special token that will only ever consist of a single pound sign.
199                    output_tokens.append(_AddressToken(delivery_address[current_position], _AddressToken.TYPE_POUND, current_position))
200                else:
201                    # Begins with hyphen, period, or forward slash,
202                    current_token = _AddressToken(delivery_address[current_position], _AddressToken.TYPE_SPECIAL, current_position)
203            else:
204                if re.match(r"[^A-Za-z0-9\-\.\/]", delivery_address[current_position]):
205                    # We encountered a non-parsed character. Terminate the current token.
206                    output_tokens.append(current_token)
207                    current_token = None
208                    
209                    # Also terminate the token if we encounter a pound (#) sign.
210                    # However, we also want to add the pound sign in as a separate token.
211                    if re.match(r"#", delivery_address[current_position]):
212                        # Special pound sign token.
213                        output_tokens.append(_AddressToken(delivery_address[current_position], _AddressToken.TYPE_POUND, current_position))
214                else:
215                    # Append whatever character we have on hand to the previous token.
216                    if current_token.type_ == _AddressToken.TYPE_NUMBER:
217                        if re.match(r"\.", delivery_address[current_position]):
218                            # A dot indicates a decimal
219                            current_token.type_ = _AddressToken.TYPE_DECIMAL
220                        elif re.match(r"\/", delivery_address[current_position]):
221                            # A slash indicates a fraction
222                            current_token.type_ = _AddressToken.TYPE_FRACTION
223                        elif re.match(r"[^0-9]", delivery_address[current_position]):
224                            # This is no longer a number, but a suffixed number
225                            current_token.type_ = _AddressToken.TYPE_NUMBERSUFFIXED
226                    elif current_token.type_ == _AddressToken.TYPE_DECIMAL or current_token.type_ == _AddressToken.TYPE_FRACTION:
227                        if re.match(r"[^0-9]", delivery_address[current_position]):
228                            # This is not a decimal or a fraction, it is now a generic suffixed number.
229                            current_token.type_ = _AddressToken.TYPE_NUMBERSUFFIXED
230                    current_token.literal += delivery_address[current_position]
231            current_position += 1
232        # Append the last token in the string, if any.
233        if current_token is not None:
234            output_tokens.append(current_token)
235        return output_tokens

377                already_seen_hashtag = True
378            if current_token.type_ == _AddressToken.TYPE_SPECIAL:
379                raise AddressParseError("unexpected token", delivery_address, current_token.index)
380        
381        # The secondary address must be after the street suffix.
382        if suffix_index is not None and address2_index is not None and address2_index <= suffix_index:
383            address2_index = None
384        
385        # Parse the secondary address and pop it from the list, if it exists
386        if address2_index is not None:
387            address2_type = unparsed_tokens[address2_index].literal
388            if address2_index == len(unparsed_tokens) - 1:
389                raise AddressParseError("missing number for secondary address", delivery_address, unparsed_tokens[-1].index + len(unparsed_tokens[-1].literal))
390            address2 = " ".join((x.literal for x in unparsed_tokens[address2_index+1:]))
391            
392            # Keep removing unparsed tokens until we have reached the index of address2.
393            while len(unparsed_tokens) > address2_index:
394                unparsed_tokens.pop()
395        
396        # Now, we have the predirectional, street name, and street suffix, and postdirectional.
397        
398        # Check if we have a postdirectional.
399        if len(unparsed_tokens) > 0 and unparsed_tokens[-1].literal.replace('.', "") in DIRECTIONS_LIST:
400            # Yes, there is a postdirectional! Indicate its presence and pop it from the list.
401            postdirectional = unparsed_tokens[-1].literal
402            unparsed_tokens.pop()
403        
404        # At this point, only the predirectional, street name, and street suffix should be in the list, if they exist at all.
405        
406        # Now check the location of the suffix. If it's not at the end, the "suffix" we identified was actually not the true suffix.
407        if suffix_index is not None and len(unparsed_tokens) - suffix_index > 1:
408            # Okay, the "suffix" that we expected turned out not the be the suffix. That's okay, let's mark it as not a suffix.
409            suffix_index = None
410        
411        # TODO Standardize highway names (like COUNTY ROAD 238A, INTERSTATE 75, etc.)
412        
413        if suffix_index is None:
414            # The starting and ending index of the street name. If there is a postdirectional, don't count that here.
415            if len(unparsed_tokens) > 1 and unparsed_tokens[0].literal.replace('.', "") in DIRECTIONS_LIST:
416                predirectional = unparsed_tokens[0].literal
417                unparsed_tokens.pop(0)
418            street_name = " ".join((x.literal for x in unparsed_tokens))
419        else:
420            street_start_index = 0
421            if suffix_index > 1:
422                if unparsed_tokens[0].literal.replace('.', "") in DIRECTIONS_LIST:
423                    predirectional = unparsed_tokens[0].literal
424                    street_start_index = 1
425            street_suffix = unparsed_tokens[suffix_index].literal
426            street_name = " ".join((x.literal for x in unparsed_tokens[street_start_index:suffix_index]))
427        
428        # More standardization of the address.
429        if predirectional is not None:
430            predirectional = re.sub(r"[^A-Za-z]", "", predirectional)
431        if postdirectional is not None:
432            postdirectional = re.sub(r"[^A-Za-z]", "", postdirectional)
433        if street_suffix is not None:
434            street_suffix = re.sub(r"[^A-Za-z]", "", street_suffix)
435        predirectional = DIRECTIONS_MAPPING[predirectional] if predirectional in DIRECTIONS_MAPPING else predirectional
436        postdirectional = DIRECTIONS_MAPPING[postdirectional] if postdirectional in DIRECTIONS_MAPPING else postdirectional
437        street_suffix = STREET_SUFFIX_MAPPING[street_suffix] if street_suffix in STREET_SUFFIX_MAPPING else street_suffix
438        
439        # Hyphens will be replaced with spaces by USPS as well.
440        street_name = street_name.replace('-', ' ')
441        
442        return StandardAddress(address_number, predirectional, street_name, street_suffix, postdirectional, address2_type, address2,
443                city, state, zipcode, zipcode_ext)
